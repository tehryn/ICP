143 void G_Board::process_command()
144 {
145  if (was_hint)
146  unset_hint();
147 
148  switch (from_type) {
149  case Hidden:
150  game.fromH_toV();
151  rebuild_stack(from_type,0);
152  rebuild_stack(Visible, 0);
153  break;
154 
155  case Visible:
156  Card_deck_visible.set_border("border: 0px;", Card_deck_visible.top());
157  if (to_type == Single_Color)
158  {
159  if (!game.fromV_toC(to_id))
160  {}//std::cerr << "Unsuccessful command." << std::endl;
161  else
162  {
163  rebuild_stack(Visible, 0);
164  rebuild_stack(Single_Color, to_id);
165  }
166  }
167  else if (to_type == Working)
168  {
169  if (!game.fromV_toW(to_id))
170  {}//std::cerr << "Unsuccessful command." << std::endl;
171  else
172  {
173  rebuild_stack(Visible, 0);
174  rebuild_stack(Working, to_id);
175  }
176  }
177  else
178  {}//std::cerr << "Unknown command." << std::endl;
179  break;
180 
181  case Single_Color:
182  Single_color_stack[from_id].set_border("border: 0px;", Single_color_stack[from_id].top());
183  if (to_type == Single_Color)
184  {
185  if (!game.fromC_toC(from_id, to_id))
186  {}//std::cerr << "Unsuccessful command." << std::endl;
187  else
188  {
189  rebuild_stack(Single_Color, from_id);
190  rebuild_stack(Single_Color, to_id);
191  }
192  }
193  else if (to_type == Working)
194  {
195  if (!game.fromC_toW(from_id, to_id))
196  {}//std::cerr << "Unsuccessful command." << std::endl;
197  else
198  {
199  rebuild_stack(Single_Color, from_id);
200  rebuild_stack(Working, to_id);
201  }
202  }
203  else
204  {}//std::cerr << "Unknown command." << std::endl;
205  break;
206 
207  case Working:
208  Working_stack[from_id].set_border("border: 0px;", clicked);
209  if (to_type == Single_Color)
210  {
211  if (!game.fromW_toC(from_id, to_id))
212  {}//std::cerr << "Unsuccessful command." << std::endl;
213  else
214  {
215  rebuild_stack(Working, from_id);
216  rebuild_stack(Single_Color, to_id);
217  }
218  }
219  else if (to_type == Working)
220  {
221  if (!game.fromW_toW(from_id, to_id, clicked->card))
222  {}//std::cerr << "Unsuccessful command." << std::endl;
223  else
224  {
225  rebuild_stack(Working, from_id);
226  rebuild_stack(Working, to_id);
227  }
228  }
229  else
230  {}//std::cerr << "Unknown command." << std::endl;
231  break;
232 
233  default:
234  break;
235  }
236  // writing score to board
237  str_score = "Score:\n";
238  str_score += std::to_string(game.get_score());
239  if (game.is_victory())
240  str_score += "\nVictory";
241  score.setText(str_score.c_str());
242 
243  proc_move = false;
244 }

395 void G_Board::hint()

404  Working_stack[hint_move.get_from()].set_border(hint_from_border, hint_move.get_card());

406  break;
407

410  Single_color_stack[hint_move.get_to()].set_border(hint_to_border, Single_color_stack[hint_move.get_to()].top());
411  break;
412 
413  case VC:
414  Card_deck_visible.set_border(hint_from_border, Card_deck_visible.top());
415  Single_color_stack[hint_move.get_to()].set_border(hint_to_border, Single_color_stack[hint_move.get_to()].top());
416  break;
417 
418  case VW:
419  Card_deck_visible.set_border(hint_from_border, Card_deck_visible.top());
420  Working_stack[hint_move.get_to()].set_border(hint_to_border, Working_stack[hint_move.get_to()].top());
421  break;

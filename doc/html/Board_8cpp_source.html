281  Working_stack new_stack;

283  // gets data from last move and undo it

293  for(unsigned i = 0; i < size; i++) {
294  working_stacks[move.get_from()].force_push(new_stack.pop_bottom());
295  }
296  break;
297  case WC:
298  if (move.was_turned()) {
299  working_stacks[move.get_from()].set_top_hidden();
300  }
301  tmp = color_stacks[move.get_to()].pop();
302  working_stacks[move.get_from()].force_push(tmp);

305  case CW:
306  tmp = working_stacks[move.get_to()].pop();
307  color_stacks[move.get_from()].force_push(tmp);
308  score += 15;
309  break;
310  case CC:

312  color_stacks[move.get_from()].force_push(tmp);
313  break;
314  case VW:
315  tmp = working_stacks[move.get_to()].pop();
